        // Main function to show information about a business structure

        function showInfo(type) {

            const state = stateSelect.value;

            if (stateData[state] && stateData[state][type]) {

                const info = stateData[state][type];

                infoDisplay.innerHTML = `

                    <h3 class="text-xl md:text-2xl font-bold text-gray-800 mb-2">${info.title}</h3>

                    ${info.content}

                `;

            } else {

                infoDisplay.innerHTML = `<p class="font-medium text-red-600">Information not available for this state yet. Please select another state or add the information yourself by editing the code!</p>`;

            }

        }

        function showCurrentStateInfo() {

            showInfo(stateSelect.value);

        }

        // Holding company functions

        function createParentCompany() {

            const name = document.getElementById('parent-name').value;

            const type = document.getElementById('parent-type').value;

            const state = stateSelect.value;

            if (name && type) {

                parentCoName.textContent = name;

                parentCoType.textContent = type;

                parentDisplay.classList.remove('hidden');

                addSubSection.classList.remove('hidden');

                infoDisplay.innerHTML = `<p class="font-medium text-blue-800">Parent company "${name}" (${type}) in ${stateData[state].name} created! Now add your subsidiary businesses below.</p>`;

            } else {

                infoDisplay.innerHTML = `<p class="font-medium text-red-600">Please enter a name and select a structure for your parent company.</p>`;

            }

        }

        function addSubsidary() {

            const parentName = document.getElementById('parent-name').value;

            if (!parentName) {

                infoDisplay.innerHTML = `<p class="font-medium text-red-600">Please create a parent company first.</p>`;

                return;

            }

            const subName = document.getElementById('sub-name').value;

            const subType = document.getElementById('sub-type').value;

            if (subName && subType) {

                subsTitle.classList.remove('hidden');

                const subItem = document.createElement('div');

                subItem.className = 'bg-gray-200 border-l-4 border-gray-400 p-3 rounded-lg flex justify-between items-center';

                subItem.innerHTML = `

                    <span class="text-gray-800 font-medium">${subName} (${subType})</span>

                    <button onclick="removeSubsidiary(this)" class="text-red-500 hover:text-red-700 transition-colors">

                        <svg xmlns="http://www.w3.org/2000/svg" class="h-5 w-5" viewBox="0 0 20 20" fill="currentColor">

                            <path fill-rule="evenodd" d="M9 2a1 1 0 00-.894.553L7.382 4H4a1 1 0 000 2v10a2 2 0 002 2h8a2 2 0 002-2V6a1 1 0 100-2h-3.382l-.724-1.447A1 1 0 0011 2H9zM7 8a1 1 0 012 0v6a1 1 0 11-2 0V8zm5-1a1 1 0 00-1 1v6a1 1 0 102 0V8a1 1 0 00-1-1z" clip-rule="evenodd" />

                        </svg>

                    </button>

                `;

                subsidiaryList.appendChild(subItem);

               

                document.getElementById('sub-name').value = ''; // Clear input

                document.getElementById('sub-type').value = ''; // Clear selection

               

                infoDisplay.innerHTML = `<p class="font-medium text-green-600">Added new subsidiary: "${subName}" (${subType}).</p>`;

            } else {

                infoDisplay.innerHTML = `<p class="font-medium text-red-600">Please enter a name and select a structure for the subsidiary.</p>`;

            }

        }

        // Function to remove a subsidiary

        function removeSubsidiary(button) {

            button.parentElement.remove();

            infoDisplay.innerHTML = `<p class="font-medium text-gray-600">Subsidiary removed.</p>`;

            if (subsidiaryList.childElementCount === 0) {

                subsTitle.classList.add('hidden');

            }

        }
